Permutations and combinations of strings

(def chars-to-play-with (seq (char-array "asdf")))

(clojure.string/join chars-to-play-with)

(clojure.string/join '(\N \o \w))

(def short-phrase "Facetiously speaking, what is special about May the fourth?")

(def up-case-word-list (map clojure.string/upper-case (clojure.string/split short-phrase #" ")))

(clojure.string/replace (nth up-case-word-list 1) #"," "")

(clojure.string/replace (nth up-case-word-list 8) #"[\?\,]" "")

(for [str up-case-word-list] (clojure.string/replace str #"[\,\?]" ""))

(require '(clojure [set :as s]))

(apply s/union (map set (for [str up-case-word-list] (clojure.string/replace str #"[\,\?]" ""))))

(def unique-char-set
  (apply s/union (map set (map (fn [str] (clojure.string/replace str #"[\,\?]" "")) up-case-word-list))))

(count unique-char-set)
