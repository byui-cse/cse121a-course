(defn decode-prime-list
  ([elements] (decode-prime-list elements 2 nil)) ; facade clause
  ([elements counter accumulator] ; working clause
   (if (= (first elements) 0) ; if the first element of the list is 0
     (def inc 1);set the incrementor to be 1
     (def inc (first elements))) ; otherwise set it to be the value of the first element
   (when (= (first elements) 1) ; when the first element of the list is 1
     (def prepended-accumulator (conj accumulator counter)));prepend the counter to the list
   (if (empty? (rest elements)) ; if the rest of the list is empty
     (reverse prepended-accumulator) ; the list was prepended, so reverse the list to get the right order and return the result
     (recur (rest elements) (+ inc counter) prepended-accumulator)))) ; otherwise, call decode-prime-list again
